# Structs


Arrays hold multiple variables of 1 data type.
Structs hold multiple variables of multiple data types.
â—‡ Usually they're used to records about stuff - e.g. books

struct Books book1; - Declare struct variable
book1.book_id - Access struct member

#include <stdio.h>
#include <string.h>

struct Books {
    int  book_id;
    char *title;
    char author[50];            // have to usr strncpy() to write into char[] in struct
    char alphabet_section;
    float rating;
} book3, book4;                 // OPTIONAL - can declare struct variables here

int main() 
{
    struct Books book1;         // declare struct variable
    struct Books book2;

    book1.book_id = 44;
    book1.title = "Do Androids Dream of Electric Sheep";
    strncpy(book1.author, "Philip K. Dick", strlen("Philip K. Dick"));
    book1.alphabet_section = 'A';
    book1.rating = 8.5;

    printf("book1.book_id          = %d \n", book1.book_id);
    printf("book1.title            = %s \n", book1.title);
    printf("book1.author           = %s \n", book1.author);
    printf("book1.alphabet_section = %c \n", book1.alphabet_section);
    printf("book1.rating           = %f \n", book1.rating);

    return 0;
}

## Pointers to Structs


struct Books *book1; - Declare struct variable as pointer
book1->book_id - Access struct member (when the struct variable is a pointer)

#include <stdio.h>
#include <string.h>

struct Books {
    int  book_id;
    char *title;
    char author[50];
    char alphabet_section;
    float rating;
} *book3, *book4;                 // OPTIONAL - can declare struct variables here

int main() 
{
    struct Books *book1;         // declare struct variable as pointer

    book1->book_id = 44;         // reference members of the struct with ->

    return 0;
}


## Nested Structs


#include <stdio.h>
#include <string.h>

struct EmployeeInfo {
    char *name;
    int   id;
};

struct EmployeeList {
    struct EmployeeInfo e1;
    struct EmployeeInfo e2;
} IT_dept, HR_dept;

int main() 
{
    IT_dept.e1.name = "Timmy Toucan";
    IT_dept.e1.id   = 44;
    
    HR_dept.e1.name = "Ginny Giraffe";
    HR_dept.e1.id   = 101;

    return 0;
}
