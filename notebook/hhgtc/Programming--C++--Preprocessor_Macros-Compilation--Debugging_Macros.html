#Debugging Macros


It's sometimes helpful to use preprocessor directives to assist with debugging code.

Use a DEBUG macro to easily turn/off debugging.
// define debug macro
#define DEBUG 1


Use the ASSERT macro function to evaluate a condition for a true/false value.
#if(DEBUG == 1)
    #define ASSERT(expression)                          \
    cout << #expression << " ... " << num;              \
    if(expression != true)                              \
    {                                                   \
        cout << " Fails in file: " << __FILE__;         \
        cout << " at line: " << __LINE__ << endl;       \
    }                                                   \
    else                                                \
    {                                                   \
        cout << " Succeeds" << endl;                    \
    }
// if debugging is off
#elif(DEBUG == 0)


Multiple statements can be included in the macro function definition by adding a backslash \ at the end of each line.
Dont' include a \ on the last line.

Example:
// define debug macro
#define DEBUG 1

// if debugging is on
#if(DEBUG == 1)
    #define ASSERT(expression)                          \
    cout << #expression << " ... " << num;              \
    if(expression != true)                              \
    {                                                   \
        cout << " Fails in file: " << __FILE__;         \
        cout << " at line: " << __LINE__ << endl;       \
    }                                                   \
    else                                                \
    {                                                   \
        cout << " Succeeds" << endl;                    \
    }                                                   
    
// if debugging is off
#elif(DEBUG == 0)
    #define ASSERT(result)                              \
    cout << "Number is " << num << endl;                
#endif

#include <iostream>
using namespace std;

int main()
{
    int num = 9;    ASSERT(num < 10);
    num++;          ASSERT(num < 10);

    return 0;
}
