strcat needs a null terminated string.
My solution was to copy both strings into a new string.
#include <stdio.h>
#include <Windows.h>

int main()
{
// strings to concatenate
char* whoami = "bob\\bob-dev";
char* administrator = "_ADMINISTRATOR";

printf("%s \n", whoami);
printf("%s \n", administrator);

// concate into final string
char final[121] = {0};
strcat_s(final, 121, whoami);
strcat_s(final, 121, administrator);

printf("%s \n", final);

return 0;
}
