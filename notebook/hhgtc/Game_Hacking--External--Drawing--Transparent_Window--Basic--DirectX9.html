# Basic Transparent Overlay w/ DirectX9


This is a basic transparent overlay. No error handling or fancy stuff.
It uses WS_EX_TOPMOST to always be on the topmost window. Therefore it draws over other windows.
Window creation is the same as every other method.
DirectX9 is just used for drawing.

The key parts are here:
/*
Set window flags.
- WS_POPUP - removes white background & title bar from window
- WS_EX_LAYERED - improves performance drawing visual effects. layered windows can also be translucent
- WS_EX_TRANSPARENT - if LAYERED is enabled, passes mouse events to the window below. without this, you couldn't click on the game
- WS_EX_TOPMOST - always draw the window on top, meaning we can see our cheat overlay (often detected by anit cheat)
*/
DWORD dwStyle_flags = WS_POPUP;
DWORD dwStyleEx_flags = WS_EX_LAYERED | WS_EX_TRANSPARENT | WS_EX_TOPMOST;

...

MARGINS margin = { game_pos.x, game_pos.y, g_gameWidth, g_gameHeight };
DwmExtendFrameIntoClientArea(g_overlayWindow, &margin); // Set window margins. Without this, window is black
SetLayeredWindowAttributes(g_overlayWindow, RGB(0, 0, 0), 255, LWA_ALPHA); // Set the level of transparency (255 = full)

#include <Windows.h>
#include <windowsx.h>
#include <dwmapi.h>
#include <d3d9.h>
#include <d3dx9.h>
#include <stdio.h>
#include <stdbool.h>

#pragma comment(lib, "Dwmapi.lib")
#pragma comment(lib, "d3d9.lib")

// Window creation & game info
#define TARGET_WINDOW_NAME "Untitled - Notepad"
HWND g_overlayWindow = NULL;
HWND g_gameWindow = NULL;
int g_gameWidth = 0;
int g_gameHeight = 0;

// Direct3d
LPDIRECT3D9 d3d = NULL; // Pointer to our Direct3d interface
LPDIRECT3DDEVICE9 d3d_device = NULL; // Pointer to the device class
LPD3DXLINE d3dx_line = NULL;
LPD3DXFONT d3dx_font = NULL;
int font_height = 13;
D3DCOLOR d3d_white = D3DCOLOR_ARGB(255, 255, 255, 255);
D3DCOLOR d3d_red = D3DCOLOR_ARGB(255, 255, 0, 0);
D3DCOLOR d3d_green = D3DCOLOR_ARGB(255, 0, 255, 0);
D3DCOLOR d3d_orange = D3DCOLOR_ARGB(255, 255, 103, 0);
D3DCOLOR d3d_light_blue = D3DCOLOR_ARGB(255, 79, 185, 255);

// Math
typedef struct _Vector2
{
float x, y;
} vec2;

LRESULT CALLBACK WindowProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
{
switch (msg)
{
case WM_CLOSE:
DestroyWindow(hWnd); // Destroys window and LL child windows and removes window from system
break;
case WM_DESTROY:
PostQuitMessage(0); // Sends WM_QUIT to message loop to quit
break;
default:
return DefWindowProcA(hWnd, msg, wParam, lParam);
}

return 0;
}

bool CreateTransparentWindow()
{
printf("[*] Creating transparent window \n");

// Register window class
WNDCLASSEXA wc = { 0 };
wc.cbSize = sizeof(WNDCLASSEXA);
wc.style = CS_HREDRAW | CS_VREDRAW;
wc.lpfnWndProc = WindowProc;
wc.hInstance = NULL; // hInstance
wc.hCursor = LoadCursorA(NULL, IDC_ARROW);
wc.hbrBackground = (HBRUSH)COLOR_WINDOW;
wc.lpszClassName = "IdleHands";

RegisterClassExA(&wc);

// Grab handle to game window
g_gameWindow = FindWindowA(NULL, TARGET_WINDOW_NAME);

// Get game resolution size (using Client excludes border pixels and bar)
RECT game_size = { 0 };
GetClientRect(g_gameWindow, &game_size);
g_gameWidth = game_size.right - game_size.left;
g_gameHeight = game_size.bottom - game_size.top;
printf("* Game resolution: %d x %d \n", g_gameWidth, g_gameHeight);

// Get position of game window on screen
POINT game_pos = { 0 };
ClientToScreen(g_gameWindow, &game_pos);
printf("* Top left game position: (%d,%d) \n", game_pos.x, game_pos.y);

/*
Set window flags.
- WS_POPUP - removes white background & title bar from window
- WS_EX_LAYERED - improves performance drawing visual effects. layered windows can also be translucent
- WS_EX_TRANSPARENT - if LAYERED is enabled, passes mouse events to the window below. without this, you couldn't click on the game
- WS_EX_TOPMOST - always draw the window on top, meaning we can see our cheat overlay (often detected by anit cheat)
*/
DWORD dwStyle_flags = WS_POPUP;
DWORD dwStyleEx_flags = WS_EX_LAYERED | WS_EX_TRANSPARENT | WS_EX_TOPMOST;

// Create our window over the top of the game window
g_overlayWindow = CreateWindowExA(dwStyleEx_flags, wc.lpszClassName, "smokescreen", dwStyle_flags, game_pos.x, game_pos.y, g_gameWidth, g_gameHeight, NULL, NULL, NULL, NULL);

MARGINS margin = { game_pos.x, game_pos.y, g_gameWidth, g_gameHeight };
DwmExtendFrameIntoClientArea(g_overlayWindow, &margin); // Set window margins. Without this, window is black
SetLayeredWindowAttributes(g_overlayWindow, RGB(0, 0, 0), 255, LWA_ALPHA); // Set window to be transparent

// Show window and update it to ensure it's drawn correctly
ShowWindow(g_overlayWindow, SW_SHOW);
UpdateWindow(g_overlayWindow);

return true;
}

bool InitDirect3d(HWND hWnd)
{
printf("[*] Initialising Direct3d \n");

// Create direct3d interface
d3d = Direct3DCreate9(D3D_SDK_VERSION);
printf("+ Created direct3d interface \n");

// Create direct3d device
D3DPRESENT_PARAMETERS d3dpp = { 0 };
d3dpp.Windowed = TRUE;
d3dpp.SwapEffect = D3DSWAPEFFECT_DISCARD; // Discard old frames
d3dpp.BackBufferFormat = D3DFMT_A8R8G8B8;
d3dpp.BackBufferWidth = g_gameWidth;
d3dpp.BackBufferHeight = g_gameHeight;
d3dpp.hDeviceWindow = hWnd;
d3dpp.PresentationInterval = D3DPRESENT_INTERVAL_IMMEDIATE; // For Present() with VSync, use D3DPRESENT_INTERVAL_ONE
HRESULT hres = IDirect3D9_CreateDevice(d3d, D3DADAPTER_DEFAULT, D3DDEVTYPE_HAL, hWnd, D3DCREATE_HARDWARE_VERTEXPROCESSING, &d3dpp, &d3d_device);
printf("+ Created d3d device \n");

return true;
}

void CleanupDirect3d()
{
IDirect3DDevice9_Release(d3d_device);
IDirect3D9_Release(d3d);
}

void DrawLine(vec2 src, vec2 dest, float thickness, D3DCOLOR colour)
{
// Create a d3dx line
// Creating a d3dx line every time is inefficient, so only create one if we don't have one already
if (d3dx_line == NULL)
D3DXCreateLine(d3d_device, &d3dx_line);

D3DXVECTOR2 vertex_list[2];
vertex_list[0] = (D3DXVECTOR2){ src.x, src.y };
vertex_list[1] = (D3DXVECTOR2){ dest.x, dest.y };
d3dx_line->lpVtbl->SetWidth(d3dx_line, thickness);
d3dx_line->lpVtbl->Draw(d3dx_line, vertex_list, 2, colour);

return;
}

void DrawRectangle(vec2 top, vec2 bot, float thickness, D3DCOLOR colour)
{
// Calculate height of the rectangle
// abs() will always return a positive value. fabsf() is the float version.
float height = fabsf(bot.y - top.y);
float width = height / 4;

// x is at plus/minus width of box
vec2 top_left, top_right;
top_left.x = top.x - width;
top_right.x = top.x + width;
top_left.y = top.y;
top_right.y = top.y;

vec2 bot_left, bot_right;
bot_left.x = bot.x - width;
bot_right.x = bot.x + width;
bot_left.y = bot.y;
bot_right.y = bot.y;

// Draw the rectangle
DrawLine(top_left, top_right, thickness, colour);
DrawLine(top_right, bot_right, thickness, colour);
DrawLine(bot_right, bot_left, thickness, colour);
DrawLine(bot_left, top_left, thickness, colour);

return;
}

void PrintText(int x, int y, D3DCOLOR colour, const char* format, ...)
{
// Create a d3dx font if we don't have one
if (d3dx_font == NULL)
D3DXCreateFontA(d3d_device, font_height, 0, FW_NORMAL, 1, FALSE, ANSI_CHARSET, OUT_TT_PRECIS, PROOF_QUALITY, FF_DONTCARE | DEFAULT_PITCH, "Consolas", &d3dx_font);

RECT font_rect = { 0 };
font_rect.left = x;
font_rect.right = x + 1001;
font_rect.top = y;
font_rect.bottom = y + 1001;

char text[100] = { 0 };
va_list args;
va_start(args, format);
vsprintf_s(text, 100, format, args);
va_end(args);

// Draw font and align to LEFT and TOP of rectangle
d3dx_font->lpVtbl->DrawTextA(d3dx_font, NULL, text, -1, &font_rect, DT_LEFT | DT_TOP, colour);

return;
}

int main(int argc, char* argv[])
{
CreateTransparentWindow();
InitDirect3d(g_overlayWindow);

// Window message loop (also functions as cheat loop)
MSG msg = { 0 };
while (TRUE)
{
while (PeekMessageA(&msg, NULL, 0, 0, PM_REMOVE))
{
TranslateMessage(&msg);
DispatchMessageA(&msg);
}

if (msg.message == WM_QUIT)
break;

// Prep DirectX for drawing (clear buffer and create scene)
IDirect3DDevice9_Clear(d3d_device, 0, NULL, D3DCLEAR_TARGET, D3DCOLOR_ARGB(0, 0, 0, 0), 1.0f, 0);
IDirect3DDevice9_BeginScene(d3d_device);

vec2 line_start = { 50, 50 };
vec2 line_end = { 150, 150 };
DrawLine(line_start, line_end, 2.0f, d3d_green);
vec2 rect_top = { 300, 0 };
vec2 rect_bot = { 300, 200 };
DrawRectangle(rect_top, rect_bot, 3.0f, d3d_light_blue);
PrintText(75, 85, d3d_orange, "ahoy there!");

// Draw (render buffer and display rendered frame on screen)
IDirect3DDevice9_EndScene(d3d_device);
IDirect3DDevice9_Present(d3d_device, NULL, NULL, NULL, NULL);

Sleep(1);
}

// Cleanup
CleanupDirect3d();
DestroyWindow(g_overlayWindow);

return 0;
}
