• Don't do wcslen inside of your wcscat function call - you'll get a "string is not NULL terminated" error)
• For length, do (length of string 1 + length of string 2) + 1. +1 is needed for the added NULL terminator (I think)

#include <stdio.h>
#include <Windows.h>

int main()
{
// string
wchar_t luls[] = L"OMEGALULOMEGALULOMEGALUL";
int lulsLen = wcslen(luls);

// string to copy into
wchar_t much[500] = L"w0w very much ";
int muhcLen = wcslen(much);

// concatenate
int totalLen = lulsLen + muchLen;
wcscat_s(much, totalLen + 1, luls); // need +1 len for NULL terminator

return 0;
}
