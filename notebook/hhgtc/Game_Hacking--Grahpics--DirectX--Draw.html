There's a million ways to do the same thing.

## Rectangle


ID3DXLine* d3dx_line = NULL;
LPDIRECT3DDEVICE9 my_d3dDevice = NULL;

void DrawRectangle(float start_x, float start_y, float width, float height, float thickness, D3DCOLOR colour)
{
// Create a d3dx line
// Creating a d3dx line every time is inefficient, so only create one if we don't have one already
if (d3dx_line == NULL)
D3DXCreateLine(my_d3dDevice, &d3dx_line);

D3DXVECTOR2 vertex_list[5];
vertex_list[0] = (D3DXVECTOR2){ start_x, start_y };
vertex_list[1] = (D3DXVECTOR2){ start_x + width, start_y };
vertex_list[2] = (D3DXVECTOR2){ start_x + width, start_y + height };
vertex_list[3] = (D3DXVECTOR2){ start_x, start_y + height};
vertex_list[4] = (D3DXVECTOR2){ start_x, start_y };
d3dx_line->lpVtbl->SetWidth(d3dx_line, thickness);
d3dx_line->lpVtbl->Draw(d3dx_line, vertex_list, 5, colour);

return;
}


## Lines


Draw a line from xy to xy.
ID3DXLine* d3dx_line = NULL;
LPDIRECT3DDEVICE9 my_d3dDevice = NULL;

void DrawLine(vec2 src, vec2 dest, float thickness, D3DCOLOR colour)
{
// Create a d3dx line
// Creating a d3dx line every time is inefficient, so only create one if we don't have one already
if (d3dx_line == NULL)
D3DXCreateLine(my_d3dDevice, &d3dx_line);

D3DXVECTOR2 vertex_list[2];
vertex_list[0] = (D3DXVECTOR2){ src.x, src.y };
vertex_list[1] = (D3DXVECTOR2){ dest.x, dest.y };
d3dx_line->lpVtbl->SetWidth(d3dx_line, thickness);
d3dx_line->lpVtbl->Draw(d3dx_line, vertex_list, 2, colour);

return;
}
