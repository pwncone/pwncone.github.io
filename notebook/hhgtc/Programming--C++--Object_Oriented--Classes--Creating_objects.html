#Creating an object


In order to assign and retrieve data from private members of a class, special public accessor methods must be added to the class.
These are ‘setter’ methods - to assign data
And ‘getter’ methods - to retrieve data

Accessor methods are often named as the variable they address, prefixed by ‘set’ or ‘get’
e.g. setAge() and getAge()

When setter/getter methods only have 1 line, you should declare them inline e.g.
void setAge(int years) {age = years;}

#include <iostream>
#include <string>
using namespace std;

class Dog
{
private:
    // CLASS-ONLY DOG ATTRIBUTES
    int age, weight;
    string colour;
public:
    // SETTER METHODS
    void setAge(int years) {age = years;}
    void setWeight(int pounds) {weight = pounds;}
    void setColour(string dogColour) {colour = dogColour;}

    // GETTER METHODS
    int getAge() {return age;}
    int getWeight() {return weight;}
    string getColour() {return colour;}

    // PROGRAM-WIDE DOG ACTIONS
    void bark() {cout << "WOOF!" << endl;}
};

int main()
{
    Dog fido;
    fido.setAge(6);
    fido.setWeight(40);
    fido.setColour("brown");

    cout << "Fido is " << fido.getAge() << " years old." << endl;
    cout << "Fido is " << fido.getWeight() << " pounds heavy." << endl;
    cout << "Fido is a " << fido.getColour() << " dog." << endl;

    cout << endl;
    fido.bark();
    
    return 0;
}
