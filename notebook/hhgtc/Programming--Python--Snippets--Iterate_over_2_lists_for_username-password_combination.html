Use itertools.product()

import itertools

usernameFile = "usernames.txt"
passwordFile = "passwords.txt"

with open(usernameFile, 'r') as usernameList, open(passwordFile, 'r') as passwordList:
    for username, password in itertools.product(usernameList, passwordList):
        username = username.strip("\r\n")
        password = password.strip("\r\n")
        print("[*] Trying " + username + "/" + password)


OR just use readlines() to separate each line in the file into a list and iterate through the lists
usernameList = open('usernames.txt', 'r').read().splitlines()
passwordList = open('passwords.txt', 'r').read().splitlines()
