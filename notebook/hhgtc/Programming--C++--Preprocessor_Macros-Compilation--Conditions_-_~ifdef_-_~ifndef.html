#Conditions


#ifdef checks if a macro has been defined.
if macro exists, all code will be run up to an #endif

#ifndef checks if a macro hasn't been defined.
if macro doesn't exist, all code will be run up to an #endif

#undef
A previously defined macro can be removed later using #undef (so that subsequent #ifdef checks will fail)

// if BOOK is not defined
#ifndef BOOK
    #define BOOK "The Fall of the House of Usher"
#endif

#include <iostream>
using namespace std;

int main()
{
    // if BOOK is defined
    #ifdef BOOK
        cout << BOOK << endl;
    #endif

    // if AUTHOR is not defined
    #ifndef AUTHOR
        #define AUTHOR "Edgar Allan Poe"
        cout << "Author: " << AUTHOR << endl;
    #endif

    // if BOOK is defined - undefine it
    #ifdef BOOK
        #undef BOOK
    #endif

    // if BOOK is not defined
    #ifndef BOOK
        #define BOOK "The Pit and the Pendulum"
        cout << "And more stories including.. " << BOOK << endl;
    #endif
    
    return 0;
}
